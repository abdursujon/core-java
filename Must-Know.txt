import java.util.*;

public class MyClass {
 public static String grade(double[] marks){
     List<Double>allModuleMark = new ArrayList<>();
     int total = 0;
     int totalModule = 0;
     String finalGrade = null;
     for(double m: marks){
         allModuleMark.add(m);
         totalModule++;
         total += m;
     }
     double average = total / totalModule ;
     if(average < 40){
         finalGrade = "Fail";
     }
     if(average >= 40 && average <= 49){
         finalGrade = "Third-Class Honors (3rd)";
     }
     if(average >= 50 && average <= 59){
         finalGrade = "Lower Second-Class Honors (2:2)";
     }
     if(average >= 60 && average <= 69){
         finalGrade = "Upper Second-Class Honors (2:1)";
     }
     if(average >= 70){
         finalGrade = "First-Class Honors(1st)";
     }
     return finalGrade;
 }
 //1 Get marks from user
 //2. Loop through the marks and get average
 //3. Based on average return their UK style grade
  public static void main(String args[]) {
    Scanner scan = new Scanner(System.in);
    String [] parts = scan.nextLine().trim().split("\\s+");
    double [] arrayOfMarks = new double[parts.length];
    for(int i= 0; i<parts.length; i++){
        arrayOfMarks[i] = Double.parseDouble(parts[i]);
    }
    scan.close();

   System.out.println(grade(arrayOfMarks));
  }
}
